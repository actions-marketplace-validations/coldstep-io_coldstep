# Option 1 CI Runtime Guard (Go eBPF + GitHub Action) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a single open-source product—a GitHub Action plus on-runner Go agent—that loads CO-RE eBPF programs on Linux GitHub-hosted runners, observes process execution in detect mode, and writes a job summary, establishing the monolith foundation for later network prevent/LSM parity with common CI egress-guard style inputs.

**Repository state (2026-04-12):** The **nightstalker** monolith described here is **live on `main`** (agent, composite action, BPF, CI). This document is an **archaeological** implementation checklist; steps are marked **`[x]`** for traceability, not as a live backlog.

**Architecture:** A Node 20 composite-style lifecycle is implemented as a JavaScript action with `main` and `post` entrypoints (bundled with `@vercel/ncc`). The post step stops the Go daemon started in `main`. The daemon uses `github.com/cilium/ebpf` with `bpf2go`-generated loaders for BPF objects compiled by Clang. Events are emitted from a BPF ring buffer; userspace parses them and appends markdown to `$GITHUB_STEP_SUMMARY`. v1 intentionally scopes enforcement to **detect-only** for exec visibility; cgroup-based egress control is specified as Phase B behind feature flags so the first vertical slice stays testable and shippable.

**Tech Stack:** Go 1.24+, `github.com/cilium/ebpf` v0.17+, Clang/LLVM 18+, `libbpf` development headers, Node.js 20, `@actions/core`, `@vercel/ncc`, GitHub Actions workflows on `ubuntu-latest`.

```go
	return Config{Mode: m, StepSummaryPath: summary}, nil
}
```

```c
//go:build ignore

#include "vmlinux.h"
#include <bpf/bpf_helpers.h>
#include <bpf/bpf_tracing.h>
#include <bpf/bpf_core_read.h>

char LICENSE[] SEC("license") = "Dual BSD/GPL";

struct exec_event {
	__u32 pid;
	__u8 comm[16];
};

struct {
	__uint(type, BPF_MAP_TYPE_RINGBUF);
	__uint(max_entries, 1 << 24);
} events SEC(".maps");

SEC("fentry/sched_process_exec")
int BPF_PROG(handle_sched_process_exec, struct task_struct *task, struct linux_binprm *bprm)
{
	struct exec_event *ev;

	ev = bpf_ringbuf_reserve(&events, sizeof(*ev), 0);
	if (!ev)
		return 0;

	ev->pid = (__u32)(bpf_get_current_pid_tgid() >> 32);
	bpf_get_current_comm(&ev->comm, sizeof(ev->comm));

	bpf_ringbuf_submit(ev, 0);
	return 0;
}
```

Create `internal/agent/agent.go`:

```go
package agent

import (
	"bytes"
	"context"
	"encoding/binary"
	"errors"
	"fmt"
	"log"
	"os/signal"
	"syscall"

	"github.com/cilium/ebpf/link"
	"github.com/cilium/ebpf/ringbuf"
	"github.com/dumper5000/ci-runtime-guard/internal/bpf/traceexec"
	"github.com/dumper5000/ci-runtime-guard/internal/config"
	"github.com/dumper5000/ci-runtime-guard/internal/report"
)

type execEvent struct {
	PID  uint32
	Comm [16]byte
}

func Run(ctx context.Context, cfg config.Config) error {
	if cfg.Mode == config.ModePrevent {
		return fmt.Errorf("prevent mode not implemented in v1; use detect")
	}

	var objs traceexec.TraceexecObjects
	if err := traceexec.LoadTraceexecObjects(&objs, nil); err != nil {
		return fmt.Errorf("load bpf objects: %w", err)
	}
	defer objs.Close()

	lnk, err := link.AttachTracing(link.TracingOptions{
		Program: objs.HandleSchedProcessExec,
	})
	if err != nil {
		return fmt.Errorf("attach fentry sched_process_exec: %w", err)
	}
	defer lnk.Close()

	rd, err := ringbuf.NewReader(objs.Events)
	if err != nil {
		return fmt.Errorf("ringbuf reader: %w", err)
	}
	defer rd.Close()

	log.Println("ci-runtime-guard: tracing exec (detect mode)")

	for {
		select {
		case <-ctx.Done():
			return ctx.Err()
		default:
		}

		record, err := rd.Read()
		if err != nil {
			if errors.Is(err, ringbuf.ErrClosed) {
				return nil
			}
			if ctx.Err() != nil {
				return ctx.Err()
			}
			log.Printf("ringbuf read: %v", err)
			continue
		}

		var ev execEvent
		if err := binary.Read(bytes.NewReader(record.RawSample), binary.LittleEndian, &ev); err != nil {
			log.Printf("decode: %v", err)
			continue
		}

		comm := string(bytes.TrimRight(ev.Comm[:], "\x00"))
		line := fmt.Sprintf("- pid=%d comm=%s\n", ev.PID, comm)
		if err := report.AppendJobSummary(cfg.StepSummaryPath, line); err != nil {
			log.Printf("summary: %v", err)
		}
	}
}

func Main() error {
	cfg, err := config.LoadFromEnv()
	if err != nil {
		return err
	}

	ctx, cancel := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer cancel()

	if err := Run(ctx, cfg); err != nil && !errors.Is(err, context.Canceled) {
		return err
	}
	return nil
}
```

**Note:** After `go generate`, open `traceexec_bpfel.go` and confirm exported names. With prefix `Traceexec`, `bpf2go` emits `type TraceexecObjects struct`, `LoadTraceexecObjects`, and `(*TraceexecObjects).Close`. The program field comes from the BPF function name `handle_sched_process_exec` → `HandleSchedProcessExec`. The map `events` → `Events`. If any name differs, adjust `agent.go` to match the generated file exactly.

```typescript
import * as core from '@actions/core';
import { execFileSync, spawn } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';

async function run(): Promise<void> {
  const mode = core.getInput('mode') || 'detect';
  const actionPath = process.env.GITHUB_ACTION_PATH || process.cwd();
  const pidFile = path.join(actionPath, '.ci-runtime-guard.pid');
  const binPath = path.join(actionPath, 'bin', 'ci-runtime-guard');

  execFileSync('go', ['build', '-o', binPath, './cmd/ci-runtime-guard'], {
    cwd: actionPath,
    stdio: 'inherit',
  });

  const child = spawn('sudo', ['-E', binPath, 'run'], {
    cwd: actionPath,
    env: { ...process.env, CI_GUARD_MODE: mode },
    detached: true,
    stdio: 'ignore',
  });
  child.unref();
  fs.writeFileSync(pidFile, String(child.pid), 'utf8');
  core.info(`ci-runtime-guard started pid=${child.pid}`);
}

run().catch((e) => {
  core.setFailed(e instanceof Error ? e.message : String(e));
});
```

```typescript
import * as core from '@actions/core';
import * as fs from 'fs';
import * as path from 'path';

async function post(): Promise<void> {
  const actionPath = process.env.GITHUB_ACTION_PATH || process.cwd();
  const pidFile = path.join(actionPath, '.ci-runtime-guard.pid');
  if (!fs.existsSync(pidFile)) {
    core.warning('pid file missing; agent may not have started');
    return;
  }
  const pid = Number(fs.readFileSync(pidFile, 'utf8').trim());
  try {
    process.kill(pid, 'SIGTERM');
  } catch (e) {
    core.warning(`failed to signal pid ${pid}: ${e}`);
  }
}

post().catch((e) => {
  core.setFailed(e instanceof Error ? e.message : String(e));
});
```
